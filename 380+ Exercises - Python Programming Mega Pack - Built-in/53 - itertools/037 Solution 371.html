import itertools


def pairwise(iterable):
    a, b = itertools.tee(iterable)
    next(b)
    return zip(a, b)

print(f'[3, 5, 1, 8, 9, 4] -> {list(pairwise([3, 5, 1, 8, 9, 4]))}')
print(f"['a', 'b', 'c', 'd', 'e'] -> {list(pairwise(['a', 'b', 'c', 'd', 'e']))}")
print(f"'Python' -> {list(pairwise('Python'))}")
